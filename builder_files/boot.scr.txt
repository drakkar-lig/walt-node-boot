# detect walt node type
if test "$cpu" = "arm1176"
then
    setexpr isplus sub "[^+]*" "" "${board_name}"
    if test "$isplus" = ""
    then
        setenv node_type "rpi-b"
    else
        setenv node_type "rpi-b-plus"
    fi
else
    setexpr modelnum sub ".*([23]).*" "\\1" "${board_name}"
    setenv node_type "rpi-${modelnum}-b"
fi

# dhcp command should not autostart tftp
setenv autoload no

# the node type is passed to the DHCP server using the VCI field
# (vendor class identifier)
setenv bootp_vci "$node_type"

# dhcp: get our ip address and the one of next server (in variable serverip)
dhcp || reset

# retrieve and execute the second-stage script
tftp ${scriptaddr} ${serverip}:boot/${node_type}.uboot || reset
source ${scriptaddr} || reset
